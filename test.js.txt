const fs = require('fs');
const path = './attempt.txt';

if (!fs.existsSync(path)) {
    // First attempt: Create the file and fail intentionally
    fs.writeFileSync(path, 'attempted');
    console.log('Simulating a flaky failure on Attempt 1...');
    process.exit(1); 
} else {
    // Second attempt: Clean up the file and pass!
    fs.unlinkSync(path);

}
